package ohtu.intjoukkosovellus;

public class IntJoukko {

    public final static int OLETUSKAPASITEETTI = 5, // aloitustalukon koko, positiivinen luku
            OLETUSKASVATUS = 5;  // luotava uusi taulukko on näin paljon isompi kuin vanha, positiivinen luku

    private int kasvatusluku = OLETUSKASVATUS;     // Uusi taulukko on tämän verran vanhaa suurempi.
    private int[] lukujono = new int[OLETUSKAPASITEETTI];      // Joukon luvut säilytetään taulukon alkupäässä. 
    private int alkioidenLkm = 0;    // Tyhjässä joukossa alkioiden määrä on nolla. 

    public IntJoukko() {
    }

    public IntJoukko(int kapasiteetti) {
        lukujono = new int[kapasiteetti];
    }

    public IntJoukko(int kapasiteetti, int kasvatusluku) {
        lukujono = new int[kapasiteetti];
        this.kasvatusluku = kasvatusluku;
    }

    public boolean lisaaLukuJoukkoon(int luku) {
        if (lukuKuuluuJoukkoon(luku)) return false;
        lukujono[alkioidenLkm++] = luku;
        if (alkioidenLkm == lukujono.length) {
            int[] vanhaLukujono = lukujono;
            lukujono = new int[alkioidenLkm + kasvatusluku];
            lukujono = kopioiTaulukko(vanhaLukujono, lukujono);
        }
        return true;
    }

    public boolean lukuKuuluuJoukkoon(int luku) {
        for (int i = 0; i < alkioidenLkm; i++) {
            if (luku == lukujono[i]) {
                return true;
            }
        }
        return false;
    }

    public boolean poistaLukuJoukosta(int luku) {
        for (int i = 0; i < alkioidenLkm; i++) {
            if (luku == lukujono[i]) { // luku löytyy indeksistä i
                for (int j = i; j < alkioidenLkm - 1; j++) {
                    int apu = lukujono[j];
                    lukujono[j] = lukujono[j + 1];
                    lukujono[j + 1] = apu;
                }
                alkioidenLkm--;
                return true;
            }
        }
        return false;
    }

    private int[] kopioiTaulukko(int[] vanha, int[] uusi) {
        for (int i = 0; i < uusi.length; i++) {
            if (i < vanha.length) {
                uusi[i] = vanha[i];
            }
        }
        return uusi;
    }

    public int getAlkioidenLkm() {
        return alkioidenLkm;
    }

    public int[] getLukujono() {
        return lukujono;
    }

    @Override
    public String toString() {
        if (alkioidenLkm == 0) {
            return "{}";
        }
        if (alkioidenLkm == 1) {
            return "{" + lukujono[0] + "}";
        }
        String tuotos = "{";
        for (int i = 0; i < alkioidenLkm - 1; i++) {
            tuotos += lukujono[i];
            tuotos += ", ";
        }
        tuotos += lukujono[alkioidenLkm - 1] + "}";
        return tuotos;
    }

    public int[] toIntArray() {
        int[] taulu = new int[alkioidenLkm];
        for (int i = 0; i < taulu.length; i++) {
            taulu[i] = lukujono[i];
        }
        return taulu;
    }

    public static IntJoukko yhdiste(IntJoukko a, IntJoukko b) {
        IntJoukko joukko = new IntJoukko();
        for (int i = 0; i < a.getAlkioidenLkm(); i++) {
            joukko.lisaaLukuJoukkoon(a.getLukujono()[i]);
        }
        for (int i = 0; i < b.getAlkioidenLkm(); i++) {
            joukko.lisaaLukuJoukkoon(b.getLukujono()[i]);
        }
        return joukko;
    }

    public static IntJoukko leikkaus(IntJoukko a, IntJoukko b) {
        IntJoukko joukko = new IntJoukko();
        for (int i = 0; i < a.getAlkioidenLkm(); i++) {
            for (int j = 0; j < b.getAlkioidenLkm(); j++) {
                if (a.getLukujono()[i] == b.getLukujono()[j]) {
                    joukko.lisaaLukuJoukkoon(b.getLukujono()[j]);
                }
            }
        }
        return joukko;

    }

    public static IntJoukko erotus(IntJoukko a, IntJoukko b) {
        IntJoukko joukko = new IntJoukko();
        for (int i = 0; i < a.getAlkioidenLkm(); i++) {
            joukko.lisaaLukuJoukkoon(a.getLukujono()[i]);
        }
        for (int i = 0; i < b.getAlkioidenLkm(); i++) {
            joukko.poistaLukuJoukosta(b.getLukujono()[i]);
        }
        return joukko;
    }

}
